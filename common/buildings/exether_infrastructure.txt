building_roads = {
	building_group = bg_private_infrastructure
	texture = "gfx/interface/icons/building_icons/road.dds"
	ai_nationalization_desire = 0.5 #same reluctance as railways, both will likely need subsidies to run. value is in the infra anyways
	production_method_groups = {
		pmg_base_road_quality
		pmg_road_vehicles
		pmg_road_security
	}
	should_auto_expand = { 
		occupancy >= 0.8
		OR = {
			cash_reserves_ratio > 0.5
			is_subsidized = yes
		}
		OR = {
			state = {
				sg:transportation = { 
					state_goods_pricier > 0.5 
				}
			}
			state.market_access < 1
		}
		NOT = { is_under_construction = yes }
	}
	required_construction = construction_cost_low
	ownership_type = self
		background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_farming.dds"
}
